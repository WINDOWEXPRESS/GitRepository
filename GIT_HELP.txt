git init     		-> INICIAR DIRECTORIO LOCAL
git add [FICHERO]  	-> ANIADIR FICHERO A STAGE
git add .    		-> ANIADIR TODAS LAS CARPETAS O FICHERO 
git status  		-> VISUALIZAR ESTADO DE DIRECTORIO MAS ESPECIFICO
git status -s  		-> VISUALIZAR ESTADO DE DIRECTORIO MAS SIMPLE
git log  	        -> HISOTRIAL DE LOS COMMIT MAS ESPECIFICO
git log  --oneline	-> HISOTRIAL DE LOS COMMIT MAS SIMPLE
git commit -am "" 	-> COMMIT Y A(ADD) M(MESSENGER) PARA HACER UNA DESCRIPCION DE LA SUBIDA A STAGE
git commit --amend 	-> UN EDITO DE VIM HAY MUCHAS FUNCIONES
	:i 		-> EDITAR LA DESCRIPCION DE LA SUBIDA
	:wq		-> FIJAR Y SALIR DE VIM

git pull 		-> BAJAR DE REMOTO A LOCAL

git push --tags		-> SUBIR A REMOTO LOS TAGS
git clone url 		-> CLONAR REPOSITORIO REMOTO A LOCAL
